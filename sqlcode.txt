CREATE TABLE "Медведи" (
	"Возраст"	INTEGER,
	"Вид"	TEXT,
	"Место Рождения"	TEXT,
	"Дата рождения"	REAL,
	"ID поставщика"	INTEGER,
	"ID"	INTEGER UNIQUE,
	"Цена"	TEXT,
	"ID условий размещения"	INTEGER,
	FOREIGN KEY("ID условий размещения") REFERENCES "Условия содержания"("ID"),
	FOREIGN KEY("ID поставщика") REFERENCES "Поставщики"("ID"),
	PRIMARY KEY("ID" AUTOINCREMENT)
);
CREATE TABLE "Персонал" (
	"ФИО"	TEXT,
	"Возраст"	INTEGER,
	"Зар. плата"	BLOB,
	"Пол"	TEXT,
	"Дата рождения"	TEXT,
	"Место проживания"	TEXT
);
CREATE TABLE "Поставщики" (
	"ID"	INTEGER,
	"Расположение"	INTEGER,
	"Расценки"	TEXT,
	"Кол-во прошлых закупок"	INTEGER,
	"Кол-во товара в наличии"	INTEGER,
	PRIMARY KEY("ID" AUTOINCREMENT)
);
CREATE TABLE "Условия содержания" (
	"ID"	INTEGER,
	"Температура(с)"	INTEGER,
	"Требуемая пища"	TEXT,
	"Требуемое пространство(м^3)"	INTEGER,
	"Необходимость наличия других особей"	TEXT,
	PRIMARY KEY("ID" AUTOINCREMENT)
);
